                {
                  description: 'Images',
                  accept: {
                    'image/*': ['.png', '.gif', '.jpeg', '.jpg']
                  }
                },
              ],
              excludeAcceptAllOption: true,
              multiple: false
            };

            async function getTheFile() {
              // open file picker
              [fileHandle] = await window.showOpenFilePicker(pickerOpts);

              // get file contents
              const fileData = await fileHandle.getFile();
